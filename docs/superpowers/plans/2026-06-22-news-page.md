> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 wwwsite 站内新增"动态"模块（公告/公司动态场景），admin 可发布，前台用户通过 nav"动态"链接查看。

**Architecture:** 新建 `news` 表（与 `product_docs` 同结构但无 product_id）+ 9 个 db.js 函数 + 11 个 API endpoints（admin CRUD/publish + 公开列表/详情/限流 view）+ admin-news.html + news.html 单文件双模式 + nav 加"动态"链接。复用现有 Quill / DOMPurify / 上传 / 限流 模式。

**Tech Stack:** Node.js + Express + MySQL + mysql2/promise + Quill 1.3.7 + DOMPurify 3.0.6 + express-rate-limit + 现有 admin sidebar 模式。

## Global Constraints

- **数据模型**（精确）：
  - 表名：`news`
  - 字段：`id` / `title VARCHAR(255) NOT NULL` / `slug VARCHAR(255) NOT NULL UNIQUE` / `excerpt TEXT` / `content_html MEDIUMTEXT` / `cover_image VARCHAR(500)` / `category VARCHAR(50)` / `is_pinned BOOLEAN DEFAULT FALSE` / `status ENUM('draft','published') DEFAULT 'draft'` / `view_count INT DEFAULT 0` / `published_at DATETIME NULL` / `sort_order INT DEFAULT 0` / `created_at` / `updated_at`
  - 索引：`INDEX idx_status_published (status, published_at DESC)` + `INDEX idx_is_pinned (is_pinned)`
- **API endpoints**（精确）：
  - Admin：`GET/POST /api/admin/news` / `GET/PUT/DELETE /api/admin/news/:id` / `POST /api/admin/news/:id/publish` / `POST /api/admin/news/:id/unpublish`
  - 公开：`GET /api/news` / `GET /api/news?category=X` / `GET /api/news/:slug` / `POST /api/news/:id/view`（限流 30/min）
- **公开列表排序**：`is_pinned DESC, published_at DESC`（置顶优先 + 发布时间倒序）
- **db.js 函数签名**（精确）：
  - `getAllNews()` → `Array<Object>`（admin，draft+published）
  - `getNews(id)` → `Object|null`
  - `getNewsBySlug(slug)` → `Object|null`（公开）
  - `getPublishedNews({category, page, pageSize})` → `Array<Object>`（公开，分页）
  - `createNews(data)` → `{id}`（admin，data 含 title/slug/excerpt/content_html/cover_image/category/is_pinned/sort_order）
  - `updateNews(id, data)` → `{ok}`（admin）
  - `deleteNews(id)` → `{ok}`（admin）
  - `publishNews(id)` → `{ok}`（admin，首次设置 published_at = NOW()）
  - `unpublishNews(id)` → `{ok}`（admin，status=draft，**保留** published_at）
  - `incrementNewsView(id)` → `{ok}`（公开，原子 UPDATE view_count = view_count + 1）
- **URL 路由**：`/news.html`（列表）+ `/news.html?slug=xxx`（详情），单文件双模式
- **snake_case DB / camelCase API**：DB 列 `is_pinned`，API 字段 `isPinned`（per memory `feedback_prisma_snake_case_consistency.md`）
- **slug 校验**：`/^[a-z0-9-]+$/`（与 docs 一致）
- **slug 唯一性**：DB UNIQUE 约束 + admin 400 错误捕获
- **view_count 竞态**：原子 SQL `UPDATE ... SET view_count = view_count + 1 WHERE id = ?`
- **XSS 防御**：admin 输入 trusted 不净化；公开页详情渲染时 DOMPurify 净化
- **图片上传**：复用 `/api/upload-doc-image`（admin 端点，不新建 upload-news 端点）
- **限流**：`/api/news/:id/view` 30/min 公开限流（per memory `feedback_security_hardening.md` S12/S13）
- **Quill toolbar**：`[{header:[2,3,false]}, bold/italic/underline/strike, {list:ordered/bullet}, blockquote/code-block, link/image, clean]`
- **侧栏 nav-news**：在 `admin-sidebar.js` 的 "产品管理" group **之外**（顶级），与"概览/支持"同层，icon `megaphone`，link `/admin-news.html`
- **公开 nav "动态"**：复用现有 nav 容器，在"博客"链接**前**插入 `<a href="/news.html">动态</a>`，保留博客外链

- **DB 列 snake_case / API 字段 camelCase**（per memory `feedback_prisma_snake_case_consistency.md`）

```js
const mysql = require('mysql2/promise');
require('dotenv').config();

const TABLE_NAME = 'news';

const CREATE_SQL = `
CREATE TABLE IF NOT EXISTS news (
  id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  title VARCHAR(255) NOT NULL,
  slug VARCHAR(255) NOT NULL UNIQUE,
  excerpt TEXT,
  content_html MEDIUMTEXT,
  cover_image VARCHAR(500),
  category VARCHAR(50),
  is_pinned TINYINT(1) DEFAULT 0,
  status ENUM('draft','published') DEFAULT 'draft',
  view_count INT(11) DEFAULT 0,
  published_at TIMESTAMP NULL,
  sort_order INT(11) DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_status_published (status, published_at),
  INDEX idx_is_pinned (is_pinned)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
`;

async function main() {
  const conn = await mysql.createConnection({
    host: process.env.DB_HOST,
    user: process.env.DB_USER,
    password: process.env.DB_PASSWORD,
    database: process.env.DB_NAME
  });

  try {
    // 检查表是否已存在
    const [rows] = await conn.query(
      "SELECT TABLE_NAME FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = ? AND TABLE_SCHEMA = ?",
      [TABLE_NAME, process.env.DB_NAME]
    );
    if (rows.length > 0) {
      console.log(`[skip] news 表已存在`);
    } else {
      await conn.query(CREATE_SQL);
      console.log(`[ok] news 表已创建`);
    }
  } finally {
    await conn.end();
  }
}

main().catch(err => {
  console.error('[fail]', err.message);
  process.exit(1);
});
```

```js
// === News 模块（公告/公司动态） ===
async function getAllNews() {
  const [rows] = await pool.query(
    'SELECT * FROM news ORDER BY is_pinned DESC, published_at DESC, sort_order DESC, id DESC'
  );
  return rows;
}

async function getNews(id) {
  const [rows] = await pool.query('SELECT * FROM news WHERE id = ?', [id]);
  return rows[0] || null;
}

async function getNewsBySlug(slug) {
  const [rows] = await pool.query('SELECT * FROM news WHERE slug = ?', [slug]);
  return rows[0] || null;
}

async function getPublishedNews({ category, page = 1, pageSize = 12 } = {}) {
  const offset = (page - 1) * pageSize;
  let sql = 'SELECT id, title, slug, excerpt, cover_image, category, is_pinned, status, view_count, published_at, sort_order, created_at FROM news WHERE status = ?';
  const params = ['published'];
  if (category) {
    sql += ' AND category = ?';
    params.push(category);
  }
  sql += ' ORDER BY is_pinned DESC, published_at DESC, sort_order DESC, id DESC LIMIT ? OFFSET ?';
  params.push(pageSize, offset);
  const [rows] = await pool.query(sql, params);
  return rows;
}

async function createNews(data) {
  const { title, slug, excerpt, contentHtml, coverImage, category, isPinned = false, sortOrder = 0 } = data;
  const [result] = await pool.query(
    'INSERT INTO news (title, slug, excerpt, content_html, cover_image, category, is_pinned, sort_order) VALUES (?, ?, ?, ?, ?, ?, ?, ?)',
    [title, slug, excerpt, contentHtml, coverImage, category, isPinned ? 1 : 0, sortOrder]
  );
  return { id: result.insertId };
}

async function updateNews(id, data) {
  const { title, slug, excerpt, contentHtml, coverImage, category, isPinned, sortOrder } = data;
  await pool.query(
    'UPDATE news SET title = ?, slug = ?, excerpt = ?, content_html = ?, cover_image = ?, category = ?, is_pinned = ?, sort_order = ? WHERE id = ?',
    [title, slug, excerpt, contentHtml, coverImage, category, isPinned ? 1 : 0, sortOrder, id]
  );
  return { ok: true };
}

async function deleteNews(id) {
  await pool.query('DELETE FROM news WHERE id = ?', [id]);
  return { ok: true };
}

async function publishNews(id) {
  // 首次发布设置 published_at = NOW()；已发布则保持原值
  await pool.query(
    'UPDATE news SET status = ?, published_at = COALESCE(published_at, NOW()) WHERE id = ?',
    ['published', id]
  );
  return { ok: true };
}

async function unpublishNews(id) {
  // 撤稿：status=draft，**保留** published_at
  await pool.query('UPDATE news SET status = ? WHERE id = ?', ['draft', id]);
  return { ok: true };
}

async function incrementNewsView(id) {
  // 原子自增
  await pool.query('UPDATE news SET view_count = view_count + 1 WHERE id = ? AND status = ?', [id, 'published']);
  return { ok: true };
}
```

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
```

```html
  <link rel="stylesheet" href="/css/style.css">
  <link rel="stylesheet" href="/css/themes/theme-variables.css">
  <link rel="stylesheet" href="/css/pages/news.css?v=20260622-NEWTIMESTAMP">
  <script src="/vendor/dompurify/3.0.6/purify.min.js" defer></script>
  <style>
    body { font-family: 'Noto Sans SC', -apple-system, sans-serif; background: var(--bg, #fafafa); color: var(--text, #1a1a1a); }

<body>
  <header>
    <div class="nav-container">
      <a href="/" class="logo" id="header-logo">
        <div class="logo-icon" id="header-logo-icon">SV</div>
        <span class="logo-text" id="header-logo-text">博铭科技</span>

      <nav>

          <li><a href="/support">支持</a></li>

      </nav>
      <div class="header-actions" id="header-actions">
        <button class="header-cart" onclick="renderCartModal()">
          <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
            <circle cx="9" cy="21" r="1"></circle>
            <circle cx="20" cy="21" r="1"></circle>
            <path d="M1 1h4l2.68 13.39a2 2 0 0 0 2 1.61h19.72a2 2 0 0 0 2-1.61L23 6H6"></path>
          </svg>
          <span id="cart-badge" class="cart-badge" style="display:none;">0</span>
        </button>
        <a href="/login" class="header-login" id="header-login">登录</a>
        <a href="#" class="header-logout" id="header-logout" style="display:none;" onclick="headerLogout()">退出</a>
      </div>
    </div>
  </header>

  <main id="main-content">
    <div class="news-loading">加载中...</div>
  </main>

  <footer class="site-footer">
    <div class="footer-container">

    </div>
  </footer>

  <script>
    let currentCategory = '';
    let currentPage = 1;
    const pageSize = 12;
    let cachedList = [];

    function escapeHtml(s) {
      return String(s ?? '').replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'})[c]);
    }

    function formatDate(d) {
      if (!d) return '';
      const date = new Date(d);
      return date.getFullYear() + '-' + String(date.getMonth()+1).padStart(2,'0') + '-' + String(date.getDate()).padStart(2,'0');
    }

    function getSlug() {
      const params = new URLSearchParams(window.location.search);
      return params.get('slug');
    }

    function renderHeader() {
      // 复用 doc.html 的 header-actions 模式（cart 按钮 + login/logout）
      const user = window.currentUser || null;
      if (user) {
        document.getElementById('header-login').style.display = 'none';
        document.getElementById('header-logout').style.display = 'inline-block';
      }
    }

    async function loadList() {
      const main = document.getElementById('main-content');
      main.innerHTML = '<div class="news-loading">加载中...</div>';
      try {
        const params = new URLSearchParams();
        if (currentCategory) params.set('category', currentCategory);
        params.set('page', currentPage);
        params.set('pageSize', pageSize);
        const res = await fetch('/api/news?' + params.toString());
        if (!res.ok) throw new Error('HTTP ' + res.status);
        const list = await res.json();
        cachedList = list;
        // 加载所有统计（按分类计数，简化：本次只显示当前页）
        renderList();
      } catch (e) {
        main.innerHTML = '<div class="news-error">加载失败：' + escapeHtml(e.message) + '</div>';
      }
    }

    function renderList() {
      const main = document.getElementById('main-content');
      const pinned = cachedList.filter(n => n.is_pinned);
      const normal = cachedList.filter(n => !n.is_pinned);
      let html = '<div class="news-container">';
      html += '<div class="news-header"><h1>📢 博铭动态</h1><p>产品更新 · 公司新闻 · 安全公告</p></div>';
      // 分类 tab
      const cats = ['产品更新', '公司新闻', '安全公告'];
      html += '<div class="news-categories">';
      html += '<button class="news-category-tab ' + (currentCategory === '' ? 'active' : '') + '" onclick="setCategory(\'\')">全部</button>';
      cats.forEach(c => {
        html += '<button class="news-category-tab ' + (currentCategory === c ? 'active' : '') + '" onclick="setCategory(\'' + c + '\')">' + escapeHtml(c) + '</button>';
      });
      html += '</div>';
      if (cachedList.length === 0) {
        html += '<div class="news-empty">暂无动态</div>';
      } else {
        // 置顶卡片
        if (pinned.length > 0 && currentCategory === '') {
          const p = pinned[0];
          html += '<a href="?slug=' + encodeURIComponent(p.slug) + '" style="text-decoration: none; color: inherit;">';
          html += '<div class="news-pinned-card">';
          if (p.cover_image) html += '<div class="news-cover" style="background-image:url(' + escapeHtml(p.cover_image) + ')"></div>';
          else html += '<div class="news-cover"></div>';
          html += '<div class="news-body">';
          html += '<div class="news-meta"><span class="news-pinned-badge">📌 置顶</span>' + (p.category ? escapeHtml(p.category) + ' · ' : '') + formatDate(p.published_at) + ' · 👁 ' + (p.view_count || 0) + ' 阅读</div>';
          html += '<h2>' + escapeHtml(p.title) + '</h2>';
          if (p.excerpt) html += '<div class="news-excerpt">' + escapeHtml(p.excerpt) + '</div>';
          html += '</div></div></a>';
        }
        // 普通卡片
        html += '<div class="news-grid">';
        normal.forEach(n => {
          html += '<a href="?slug=' + encodeURIComponent(n.slug) + '" style="text-decoration: none; color: inherit;">';
          html += '<div class="news-card">';
          if (n.cover_image) html += '<div class="news-cover" style="background-image:url(' + escapeHtml(n.cover_image) + ')"></div>';
          else html += '<div class="news-cover"></div>';
          html += '<div class="news-body">';
          if (n.category) html += '<div class="news-category">' + escapeHtml(n.category) + '</div>';
          html += '<h3>' + escapeHtml(n.title) + '</h3>';
          if (n.excerpt) html += '<div class="news-excerpt">' + escapeHtml(n.excerpt) + '</div>';
          html += '<div class="news-meta">' + formatDate(n.published_at) + ' · 👁 ' + (n.view_count || 0) + ' 阅读</div>';
          html += '</div></div></a>';
        });
        html += '</div>';
        // 分页（简化：仅显示当前页号 + 上下页）
        html += '<div class="news-pagination">';
        html += '<button onclick="changePage(' + (currentPage - 1) + ')"' + (currentPage <= 1 ? ' disabled' : '') + '>‹ 上一页</button>';
        html += '<button class="active">' + currentPage + '</button>';
        html += '<button onclick="changePage(' + (currentPage + 1) + ')"' + (cachedList.length < pageSize ? ' disabled' : '') + '>下一页 ›</button>';
        html += '</div>';
      }
      html += '</div>';
      main.innerHTML = html;
    }

    function setCategory(c) {
      currentCategory = c;
      currentPage = 1;
      loadList();
    }

    function changePage(p) {
      if (p < 1) return;
      currentPage = p;
      loadList();
      window.scrollTo(0, 0);
    }

    async function loadDetail(slug) {
      const main = document.getElementById('main-content');
      main.innerHTML = '<div class="news-loading">加载中...</div>';
      try {
        const res = await fetch('/api/news/' + encodeURIComponent(slug));
        if (!res.ok) {
          main.innerHTML = '<div class="news-error">动态不存在</div>';
          return;
        }
        const n = await res.json();
        // view_count +1（fire-and-forget，不阻塞渲染）
        fetch('/api/news/' + n.id + '/view', { method: 'POST' }).catch(() => {});
        let html = '<div class="news-detail-container">';
        html += '<a href="/news.html" class="news-back-link">← 返回动态列表</a>';
        if (n.cover_image) html += '<div class="news-detail-cover" style="background-image:url(' + escapeHtml(n.cover_image) + ')"></div>';
        html += '<div class="news-detail-meta">' + (n.category ? escapeHtml(n.category) + ' · ' : '') + formatDate(n.published_at) + ' · 👁 ' + (n.view_count || 0) + ' 阅读</div>';
        html += '<h1 class="news-detail-title">' + escapeHtml(n.title) + '</h1>';
        const safeHtml = window.DOMPurify ? window.DOMPurify.sanitize(n.content_html || '', { USE_PROFILES: { html: true } }) : (n.content_html || '');
        html += '<div class="news-detail-content">' + safeHtml + '</div>';
        html += '</div>';
        main.innerHTML = html;
        document.title = n.title + ' - 博铭动态';
      } catch (e) {
        main.innerHTML = '<div class="news-error">加载失败：' + escapeHtml(e.message) + '</div>';
      }
    }

    // Init
    renderHeader();
    const slug = getSlug();
    if (slug) {
      loadDetail(slug);
    } else {
      loadList();
    }
  </script>
</body>
</html>
```
